### 서보모터 참참참
#### 링크 : https://www.tinkercad.com/things/0xgDMGV9r67
#### 코드 

'''c
#include <Servo.h>

Servo myservo;

int num_tones = 3;
int tones = 392;
int tempo = 100;

void setup() {
  myservo.attach(9);
  myservo.write(90);
  pinMode(10, INPUT_PULLUP);
  pinMode(2, OUTPUT);
}

void direction(){
  int x;

  x = random(3);

  if (x==0){
    myservo.write(20);
  }
  else if(x==1){
    myservo.write(90);
  }
  else if (x==2){
    myservo.write(160);
  }
}

void loop() {
  if (digitalRead(10)==LOW){

    for (int i =0; i< num_tones; i++){
      tone(2,tones,tempo);
      delay(1000);
    }

    direction();
    delay(3000);
    myservo.write(90);
  }
}
```

#### 느낀점
##### 서보모터를 이용하여 내가 원하는 만큼 각도를 조절할 수 있다는 것을 알게되었다.
